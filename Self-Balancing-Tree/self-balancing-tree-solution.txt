        Node insertThis = new Node();
        insertThis.val = val;
        Node prev=null;
    
          while(root != null){
              prev = root;
              if(val <= root.val){
                  root = root.left;
              }
              else{
                  root = root.right;
              }
                  
          }
          if(val > prev.val)
              prev.right = insertThis;
          else
              prev.left = insertThis;

      return newRoot;
    }

static int height(Node root) {
        
    if (root == null) {
        return -1;
    }

    int lefth = height(root.left);
    int righth = height(root.right);

    if (lefth > righth) {
        return lefth + 1;
